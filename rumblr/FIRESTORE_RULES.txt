Paste these rules into:
Firebase Console → Firestore Database → Rules → Publish

────────────────────────────────────────────────────────────

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // Admin must be the designated UID AND have verified: true in their user profile.
    // This prevents someone who gains the UID from acting as admin without verification.
    function isVerifiedAdmin() {
      return request.auth != null
          && request.auth.uid == 'wIVL4PEkOvTuBFagFAmLS1mnLEf2'
          && get(/databases/$(database)/documents/users/$(request.auth.uid)).data.get('verified', false) == true;
    }

    // Posts: anyone can read; authenticated users or AI system can create
    match /posts/{postId} {
      allow read: if true;
      allow create: if request.auth != null
                 || request.resource.data.is_ai_generated == true;
      allow update: if request.auth != null
                 && (request.auth.uid == resource.data.author_uid
                  || isVerifiedAdmin());
      allow delete: if request.auth != null
                 && (request.auth.uid == resource.data.author_uid
                  || isVerifiedAdmin());
    }

    // User profiles: anyone can read; only the account owner can write
    match /users/{uid} {
      allow read: if true;
      allow write: if request.auth != null && request.auth.uid == uid;
    }

    // Likes: anyone can read; must be logged in to write
    match /likes/{docId} {
      allow read: if true;
      allow write: if request.auth != null;
    }

    // Reposts: anyone can read; must be logged in to write
    match /reposts/{docId} {
      allow read: if true;
      allow write: if request.auth != null;
    }

    // Follows: anyone can read; only the follower can write/delete their own docs
    match /follows/{docId} {
      allow read: if true;
      allow create: if request.auth != null
                 && request.auth.uid == request.resource.data.follower_uid;
      allow delete: if request.auth != null
                 && request.auth.uid == resource.data.follower_uid;
    }

    // AI follows: anyone can read; only verified admin can write
    match /ai_follows/{docId} {
      allow read: if true;
      allow write: if isVerifiedAdmin();
    }

    // Admin collection: managed via Firebase Console only — no client access
    match /admin/{docId} {
      allow read, write: if false;
    }
  }
}
